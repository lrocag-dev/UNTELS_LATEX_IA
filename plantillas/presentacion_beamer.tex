\begin{frame}{Contexto del Proyecto}
    \begin{block}{Problema identificado}

    \end{block}
    
    \begin{alertblock}{Motivación}
        Desarrollar una solución accesible para pequeñas y medianas empresas.
    \end{alertblock}
    
    \begin{exampleblock}{Objetivo General}

    \end{exampleblock}
\end{frame}

% ============================================
\begin{frame}{Objetivos Específicos}
    \begin{enumerate}
        \item Seleccionar los componentes adecuados para el sistema

        \item Implementar el firmware para adquisición de datos
        \item Validar el sistema mediante pruebas experimentales
    \end{enumerate}
\end{frame}
