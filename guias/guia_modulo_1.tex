	\section{Introducción al Módulo}
	
	\lettrine{E}{l} presente módulo constituye la base fundamental para el dominio de \LaTeX, el sistema de composición de textos más potente y profesional utilizado en el ámbito académico y científico.
	
	A lo largo de este módulo, los estudiantes de Ingeniería Electrónica aprenderán a:
	\begin{itemize}
		\item Instalar y configurar los entornos de trabajo más utilizados
		\item Comprender la estructura básica de un documento \LaTeX
		\item Utilizar comandos esenciales para formato de texto
		\item Insertar ecuaciones matemáticas y símbolos especiales
		\item Gestionar tablas, imágenes y referencias bibliográficas
	\end{itemize}
	
	\subsection*{Objetivos del Módulo}
	\begin{itemize}
		\item Al finalizar este módulo, el estudiante será capaz de crear documentos académicos profesionales utilizando \LaTeX.

	\section{Instalación y Configuración de Entornos \LaTeX}
	
	\subsection{¿Qué es \LaTeX?}
	
	\LaTeX \ (pronunciado "Látex" o "Látek") es un sistema de composición de textos de alta calidad, utilizado especialmente para documentos científicos y técnicos. A diferencia de los procesadores de texto tradicionales (como Microsoft Word), \LaTeX \ se basa en el principio de \textbf{separar el contenido del formato}.
	
	\begin{cajateoria}
		\textbf{Ventajas de \LaTeX:}
		\begin{itemize}
			\item Ecuaciones matemáticas de calidad profesional
			\item Referencias cruzadas automáticas (figuras, tablas, ecuaciones)
			\item Gestión bibliográfica integrada
			\item Control total sobre el formato
			\item Estabilidad y compatibilidad multiplataforma
			\item Gratuito y de código abierto
		\end{itemize}
	\end{cajateoria}
	
	\subsection{Distribuciones de \LaTeX}
	
	Antes de comenzar a escribir, necesitamos instalar una \textbf{distribución} de \LaTeX, que incluye el compilador y las librerías básicas.
	
	\begin{table}[H]
		\centering
		\begin{tabular}{@{}lll@{}}
			\toprule
			\textbf{Distribución} & \textbf{Sistema operativo} & \textbf{Recomendación} \\
			\midrule
			MiKTeX & Windows & \faCheckSquare Recomendado \\
			TeX Live & Windows, Mac, Linux & Opción completa \\
			MacTeX & macOS & Para Mac \\
			\bottomrule
		\end{tabular}
		\caption{Distribuciones de \LaTeX}
		\label{tab:distribuciones}
	\end{table}
	
	\subsubsection{Instalación de MiKTeX en Windows 11}
	
	\begin{enumerate}
		\item Descargar MiKTeX desde \url{https://miktex.org}
		\item Ejecutar el instalador \codigo{basic-miktex-xxx.exe}
		\item Elegir \textbf{"Install for all users"} (requiere permisos de administrador)
		\item En \textbf{"Install missing packages on the fly"} seleccionar \textbf{"Yes"}
		\item Completar la instalación y cerrar
	\end{enumerate}
	
	\begin{cajanota}
		\textbf{Configuración importante:} Seleccionar "Yes" para la instalación automática de paquetes garantiza que nunca te falte una librería necesaria al compilar.
	\end{cajanota}
	
	\subsection{Editores de \LaTeX}
	
	Aunque podemos escribir \LaTeX \ en cualquier editor de texto, los editores especializados ofrecen ventajas como resaltado de sintaxis, autocompletado y compilación integrada.
	
	\subsubsection{Opción 1: TeXstudio (Recomendado para local)}
	
	\textbf{Ventajas:}
	\begin{itemize}
		\item Gratuito y de código abierto
		\item Resaltado de sintaxis avanzado
		item Visor de PDF integrado
		\item Asistente para ecuaciones y tablas
		\item Corrección ortográfica incluida
	\end{itemize}
	
	\textbf{Instalación:}
	\begin{enumerate}
		\item Descargar desde \url{https://texstudio.org}
		\item Ejecutar el instalador
		\item Aceptar configuración por defecto
	\end{enumerate}
	
	\subsubsection{Opción 2: Overleaf (Online)}
	
	\textbf{Ventajas:}
	\begin{itemize}
		\item No requiere instalación (funciona en navegador)
		\item Colaboración en tiempo real
		\item Control de versiones integrado
		\item Plantillas académicas predefinidas
		\item Ideal para trabajar en equipo
	\end{itemize}
	
	\textbf{Acceso:}
	\begin{enumerate}
		\item Ir a \url{https://overleaf.com}
		\item Registrarse con correo institucional
		\item Crear nuevo proyecto
	\end{enumerate}
	
	\begin{cajaejemplo}
		\textbf{Primer documento en Overleaf:}
		\begin{enumerate}
			\item Iniciar sesión en Overleaf
			\item Click en ``New Project'' → ''Blank Project''
			\item Eliminar el contenido predeterminado
			\item Escribir tu primer documento (ver sección siguiente)
			\item Click en ``Recompile'' (Ctrl+Enter)
		\end{enumerate}
	\end{cajaejemplo}

	\section{Estructura Básica de un Documento \LaTeX}
	
	\subsection{Preámbulo del Documento}
	
	Todo documento \LaTeX \ comienza con un \textbf{preámbulo}, donde se definen las características generales del documento.
	
	\begin{lstlisting}[caption=Estructura del preámbulo]
		\documentclass[12pt,letterpaper]{article}
		\usepackage[utf8]{inputenc}
		\usepackage[spanish]{babel}
		\usepackage{amsmath, amssymb}
		\usepackage{graphicx}
		
		\title{Título del Documento}
		\author{Autor del Documento}
		\date{\today}
	\end{lstlisting}
	
	\begin{cajateoria}
		\textbf{Elementos del preámbulo:}
		\begin{itemize}
			\item \comando{\textbackslash documentclass\{...\}} : Define el tipo de documento (article, report, book, beamer)
			\item \comando{\textbackslash usepackage\{...\}} : Carga librerías adicionales
			\item \comando{\textbackslash title\{...\}} : Define el título
			\item \comando{\textbackslash author\{...\}} : Define el autor
			\item \comando{\textbackslash date\{...\}} : Define la fecha
		\end{itemize}
	\end{cajateoria}
	
	\subsection{Cuerpo del Documento}
	
	El contenido del documento se escribe dentro del entorno \comando{\textbackslash begin\{document\}} y \comando{\textbackslash end\{document\}}.
	
	\begin{lstlisting}[caption=Documento completo mínimo]
		\documentclass[12pt]{article}
		\usepackage[utf8]{inputenc}
		\usepackage[spanish]{babel}
		
		\title{Mi Primer Documento}
		\author{Estudiante UNTELS}
		\date{\today}
		
		\begin{document}
			
			\maketitle
			
			\section{Introducción}
			Este es mi primer documento en \LaTeX.
			
			\section{Desarrollo}
			Aquí va el contenido principal.
			
			\section{Conclusiones}
			Aprendí a crear un documento básico.
			
		\end{document}
	\end{lstlisting}
	
	\subsection{Secciones y Capítulos}
	
	\LaTeX \ permite organizar el documento jerárquicamente:
	
	\begin{table}[H]
		\centering
		\begin{tabular}{@{}ll@{}}
			\toprule
			\textbf{Comando} & \textbf{Nivel} \\
			\midrule
			\codigo{\textbackslash part\{...\}} & Parte (máximo nivel) \\
			\codigo{\textbackslash chapter\{...\}} & Capítulo (solo en book/report) \\
			\codigo{\textbackslash section\{...\}} & Sección \\
			\codigo{\textbackslash subsection\{...\}} & Subsección \\
			\codigo{\textbackslash subsubsection\{...\}} & Subsubsección \\
			\codigo{\textbackslash paragraph\{...\}} & Párrafo \\
			\bottomrule
		\end{tabular}
		\caption{Jerarquía de secciones}
		\label{tab:secciones}
	\end{table}
	
	\begin{cajaejemplo}
		\textbf{Ejemplo de estructura jerárquica:}
		\begin{lstlisting}
			\section{Análisis de Circuitos}
			\subsection{Circuitos en Serie}
			\subsubsection{Cálculo de Resistencia Equivalente}
			En un circuito serie, la resistencia equivalente es...
		\end{lstlisting}
	\end{cajaejemplo}
	
	\subsection{Compilación del Documento}
	
	Para generar el PDF a partir del archivo \codigo{.tex}, se debe compilar:
	
	\begin{table}[H]
		\centering
		\begin{tabular}{@{}ll@{}}
			\toprule
			\textbf{Método} & \textbf{Instrucción} \\
			\midrule
			TeXstudio & Click en ``Build \& View'' (F5) \\
			Overleaf & Click en ``Recompile'' (Ctrl+Enter) \\
			Línea de comandos & \codigo{pdflatex documento.tex} \\
			\bottomrule
		\end{tabular}
		\caption{Métodos de compilación}
		\label{tab:compilacion}
	\end{table}
	
	\begin{nota}
		La primera compilación puede tardar unos segundos mientras se instalan los paquetes necesarios. Las compilaciones posteriores serán más rápidas.
	\end{nota}

	\subsection{Listas}
	
	\LaTeX \ permite crear listas ordenadas y no ordenadas:
	
	\begin{cajaejemplo}
		\textbf{Lista no ordenada (itemize):}
		\begin{lstlisting}
			\begin{itemize}
				\item Primer elemento
				\item Segundo elemento
				\item Tercer elemento
			\end{itemize}
		\end{lstlisting}
		
		\textbf{Lista ordenada (enumerate):}
		\begin{lstlisting}
			\begin{enumerate}
				\item Primer paso
				\item Segundo paso
				\item Tercer paso
			\end{enumerate}
		\end{lstlisting}
		
		\textbf{Lista con descripciones (description):}
		\begin{lstlisting}
			\begin{description}
				\item[LED] Diodo emisor de luz
				\item[Resistencia] Componente que limita corriente
				\item[Capacitor] Almacena carga eléctrica
			\end{description}
		\end{lstlisting}
	\end{cajaejemplo}
	
	\subsection{Tablas}
	
	Las tablas en \LaTeX \ se crean con el entorno \codigo{tabular}:
	
	\begin{cajaejemplo}
		\textbf{Tabla básica:}
		\begin{lstlisting}
			\begin{table}[h]
				\centering
				\caption{Componentes electrónicos}
				\begin{tabular}{|c|c|c|}
					\hline
					\textbf{Componente} & \textbf{Símbolo} & \textbf{Unidad} \\
					\hline
					Resistencia & R & $\Omega$ \\
					Capacitor & C & F \\
					Inductor & L & H \\
					\hline
				\end{tabular}
				\label{tab:componentes}
			\end{table}
		\end{lstlisting}
	\end{cajaejemplo}
	
	\begin{cajateoria}
		\textbf{Especificadores de columnas:}
		\begin{itemize}
			\item \codigo{l} : Columna alineada a la izquierda
			\item \codigo{c} : Columna centrada
			\item \codigo{r} : Columna alineada a la derecha
			\item \codigo{|} : Línea vertical entre columnas
			\item \codigo{\textbackslash hline} : Línea horizontal
		\end{itemize}
	\end{cajateoria}
	
	\subsection{Imágenes}
	
	Para insertar imágenes, se utiliza el paquete \paquete{graphicx}:
	
	\begin{cajaejemplo}
		\textbf{Inserción de imagen:}
		\begin{lstlisting}
			\usepackage{graphicx} % En el preámbulo
			
			\begin{figure}[h]
				\centering
				\includegraphics[width=0.7\textwidth]{diagrama_circuito.png}
				\caption{Diagrama de un circuito RC}
				\label{fig:circuito}
			\end{figure}
		\end{lstlisting}
	\end{cajaejemplo}
	
	\begin{nota}
		Las imágenes deben estar en la misma carpeta que el documento \codigo{.tex} o en una subcarpeta especificada con \comando{\textbackslash graphicspath\{\{figuras/\}\}}.
	\end{nota}
	
	\subsection{Referencias Bibliográficas}
	
	\LaTeX \ gestiona las referencias bibliográficas mediante BibTeX:
	
	\begin{cajaejemplo}
		\textbf{Creación del archivo \codigo{referencias.bib}:}
		\begin{lstlisting}
			@book{boylestad2022,
				title={Electrónica: Teoría de circuitos},
				author={Boylestad, Robert L.},
				year={2022},
				publisher={Pearson}
			}
			
			@article{floyd2020,
				title={Dispositivos electrónicos},
				author={Floyd, Thomas L.},
				journal={Pearson Educación},
				year={2020}
			}
		\end{lstlisting}
		
		\textbf{Cita en el documento:}
		\begin{lstlisting}
			Según \cite{boylestad2022}, la ley de Ohm establece...
			
			\bibliographystyle{ieeetr}
			\bibliography{referencias}
		\end{lstlisting}
	\end{cajaejemplo}

	\subsection{Ecuaciones en Electrónica}
	
	\begin{cajaejemplo}
		\textbf{Ejemplos de ecuaciones para Ingeniería Electrónica:}
		
		\begin{lstlisting}
			% Ley de Ohm
			\begin{equation}
				V = I \cdot R
				\label{eq:ohm}
			\end{equation}
			
			% Potencia eléctrica
			\begin{equation}
				P = V \cdot I = I^2 \cdot R = \frac{V^2}{R}
				\label{eq:potencia}
			\end{equation}
			
			% Reactancia capacitiva
			\begin{equation}
				X_C = \frac{1}{2\pi f C}
				\label{eq:xc}
			\end{equation}
			
			% Reactancia inductiva
			\begin{equation}
				X_L = 2\pi f L
				\label{eq:xl}
			\end{equation}
			
			% Impedancia en circuito RLC serie
			\begin{equation}
				Z = \sqrt{R^2 + (X_L - X_C)^2}
				\label{eq:impedancia}
			\end{equation}
			
			% Frecuencia de resonancia
			\begin{equation}
				f_0 = \frac{1}{2\pi\sqrt{LC}}
				\label{eq:resonancia}
			\end{equation}
		\end{lstlisting}
	\end{cajaejemplo}
	
	\subsection{Ecuaciones con Subíndices y Superíndices}
	
	\begin{cajaejemplo}
		\begin{lstlisting}
			% Subíndices
			$V_{in}$    % Voltaje de entrada
			$V_{out}$   % Voltaje de salida
			$R_1, R_2$  % Resistencias
			
			% Superíndices
			$x^2$       % x al cuadrado
			$10^{-3}$   % 10 elevado a -3 (mili)
			
			% Combinados
			$a_{ij}^{2}$ % a sub i,j al cuadrado
		\end{lstlisting}
		
		Resultado: $V_{in}$, $V_{out}$, $R_1, R_2$, $x^2$, $10^{-3}$, $a_{ij}^{2}$
	\end{cajaejemplo}
	
	\subsection{Ecuaciones en Varias Líneas}
	
	\begin{cajaejemplo}
		\textbf{Sistema de ecuaciones:}
		\begin{lstlisting}
			\begin{equation}
				\begin{cases}
					V_1 = I_1 R_1 + I_2 R_2 \\
					V_2 = I_2 R_2 + I_3 R_3 \\
					I_1 = I_2 + I_3
				\end{cases}
				\label{eq:sistema}
			\end{equation}
		\end{lstlisting}
		
		Resultado:
		\begin{equation}
			\begin{cases}
				V_1 = I_1 R_1 + I_2 R_2 \\
				V_2 = I_2 R_2 + I_3 R_3 \\
				I_1 = I_2 + I_3
			\end{cases}
		\end{equation}
	\end{cajaejemplo}
	
	\subsection{Matrices}
	
	\begin{cajaejemplo}
		\textbf{Matriz de admitancias:}
		\begin{lstlisting}
			\begin{equation}
				Y = \begin{bmatrix}
					Y_1 + Y_2 & -Y_2 \\
					-Y_2 & Y_2 + Y_3
				\end{bmatrix}
				\label{eq:matriz}
			\end{equation}
		\end{lstlisting}
		
		Resultado:
		\begin{equation}
			Y = \begin{bmatrix}
				Y_1 + Y_2 & -Y_2 \\
				-Y_2 & Y_2 + Y_3
			\end{bmatrix}
		\end{equation}
	\end{cajaejemplo}
	
	\subsection{Referencias Cruzadas}
	
	\begin{cajaejemplo}
		\textbf{Referenciar ecuaciones, figuras y tablas:}
		\begin{lstlisting}
			Como se muestra en la ecuación \ref{eq:ohm}, la relación...
			La Figura \ref{fig:circuito} ilustra...
			Los datos de la Tabla \ref{tab:componentes} indican...
		\end{lstlisting}
	\end{cajaejemplo}
	
	\begin{importante}
		Las referencias cruzadas requieren \textbf{dos compilaciones}: una para generar las etiquetas y otra para resolver las referencias.
	\end{importante}

		\item Una tabla con 4 filas y 3 columnas sobre valores nominales y medidos de resistencias

	\end{itemize}

	\ejercicio{4} Escribir el siguiente conjunto de ecuaciones en \LaTeX:
	\begin{itemize}

		\item Reactancia capacitiva ($X_C = 1/(2\pi f C)$)
		\item Ley de voltajes de Kirchhoff ($\sum V = 0$)

	\end{itemize}
